def getDate():
    mydate = calendar(text='', day=None, month=None, title='Calendar', width=300, height=210, timeout=None)
    day = mydate[2]
    month = Monat(mydate[1]).name
    year = mydate[0]
    theDate = ("%s.%s %s" %(day, month, year))
    return (theDate)
